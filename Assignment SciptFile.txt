--CREATE TABLE--
CREATE TABLE customer(
 cust_id CHAR(5) NOT NULL,
 cust_fname VARCHAR(30) NOT NULL,
 cust_lname VARCHAR(30) NOT NULL,
 cust_hp VARCHAR(12) NOT NULL,
 cust_email VARCHAR(50) NOT NULL,
 cust_dob DATE NOT NULL,
 cust_gender CHAR(1),
 cust_ic CHAR(12) NOT NULL,
CONSTRAINT pk_customer PRIMARY KEY (cust_id),
CONSTRAINT chk_email CHECK (REGEXP_LIKE(cust_email, '^[a-zA-Z]\w+@(\S+)$')),
CONSTRAINT chk_memberPhone CHECK (REGEXP_LIKE(cust_hp, '^0\d{9}|\d{11,12}$')),
CONSTRAINT chk_cust_ic CHECK (REGEXP_LIKE(cust_ic, '^0\d{9}|\d{12}$')),
CONSTRAINT uni_cust_ic UNIQUE (cust_ic),
CONSTRAINT chk_cust_gender CHECK (cust_gender in ('M','F'))
);

CREATE TABLE pet (
 pet_id CHAR(5) NOT NULL,
 pet_name VARCHAR(60) NOT NULL,
 pet_species VARCHAR(15) NOT NULL,
 pet_born_year DATE NOT NULL,
 cust_id CHAR(5) NOT NULL,
CONSTRAINT pk_pet PRIMARY KEY (pet_id),
CONSTRAINT fk_cust_id FOREIGN KEY (cust_id)
REFERENCES customer(cust_id)
);

CREATE TABLE payment (
 payment_id CHAR(5) NOT NULL,
 payment_amount NUMBER(7,2) NOT NULL,
 payment_date DATE NOT NULL,
 payment_type CHAR(15) NOT NULL,
CONSTRAINT pk_payment PRIMARY KEY (payment_id)
);

create table appointment
( appointment_id         char(5) NOT NULL,
  appointment_datetime date,
  appointment_status  char(10),
  pet_id CHAR(5) NOT NULL,
  payment_id CHAR(5) NULL,
  constraint pk_appointment
  primary key (appointment_id),
  constraint fk_appointment_pet
  foreign key (pet_id) references pet (pet_id),
  constraint fk_appointment_payment
  foreign key (payment_id) references payment (payment_id)
);

CREATE TABLE treatment (
 treatment_id CHAR(5) NOT NULL,
 treatment_name VARCHAR(60) NOT NULL,
 treatment_desc VARCHAR(100) NOT NULL,
 treatment_cost NUMBER(7,2) NOT NULL,
 treatment_price NUMBER(7,2) NOT NULL,
 CONSTRAINT pk_treatment PRIMARY KEY (treatment_id)
);

CREATE TABLE veterinarian(
 veterinarian_id CHAR(5) NOT NULL,
 veterinarian_name VARCHAR(60) NOT NULL,
 veterinarian_ic VARCHAR(15) NOT NULL,
 veterinarian_salary NUMBER(7,2) NOT NULL,
 veterinarian_email VARCHAR(50) NOT NULL,
 veterinarian_tel VARCHAR(12) NOT NULL,
 veterinarian_working_hours NUMBER(2) NOT NULL,
 veterinarian_status VARCHAR(12) NOT NULL,
 CONSTRAINT pk_veterinarian PRIMARY KEY (veterinarian_id),
 CONSTRAINT chk_working_hours CHECK (veterinarian_working_hours > 4 AND veterinarian_working_hours < 12),
 CONSTRAINT chk_email_2 CHECK (REGEXP_LIKE(veterinarian_email, '^[a-zA-Z]\w+@(\S+)$')),
 CONSTRAINT chk_vetePhone CHECK (REGEXP_LIKE(veterinarian_tel, '^0\d{9}|\d{11,12}$'))
);


create table cage
( cage_ID char(5) NOT NULL,
  cage_row number(3) NOT NULL,
  cage_column number(3) NOT NULL,
  constraint pk_cage
  primary key (cage_ID)
);

create table medicine
( medicine_ID char(5) NOT NULL,
  medicine_name  varchar(20)     NOT NULL,
  medicine_desc varchar(150),
  medicine_qty number (4) NOT NULL,
  medicine_price number(7,2)     NOT NULL,
  medicine_cost  number(6,2)     NOT NULL,
  constraint pk_medicine
  primary key (medicine_ID)
);

create table supplier
( supplier_id CHAR(5) NOT NULL,
  supplier_name varchar(30),
  supplier_businessid INT,
  supplier_contactno varchar(10),
 supplier_email varchar(30) ,
  constraint pk_supplier primary key (supplier_id)
);

create table supplier_detail
( supplier_id CHAR(5) NOT NULL,
  medicine_id CHAR(5) NOT NULL,
  order_date DATE NOT NULL,
  arrive_date DATE,
  medicine_qty INT,
  constraint pk_supp primary key (supplier_id , medicine_id, order_date),
  constraint fk_supplier foreign key (supplier_id) references supplier (supplier_id),
  constraint fk_medicine foreign key (medicine_id) references medicine (medicine_id)
);

CREATE TABLE appointment_details  (
 treatment_id CHAR(5) NOT NULL,
 appointment_id CHAR(5) NOT NULL,
 medicine_id CHAR(5) NOT NULL,
 medicine_qty NUMBER(3) NOT NULL,
 medicine_unitPrice NUMBER(5,2) NOT NULL,
 veterinarian_id CHAR(5) NOT NULL,
 cage_id CHAR(5) NOT NULL,
 CONSTRAINT pk_appointment_details PRIMARY KEY (treatment_id, appointment_id),
 CONSTRAINT fk_treatment_id FOREIGN KEY (treatment_id) 
 REFERENCES treatment(treatment_id),
 CONSTRAINT fk_appointment_id FOREIGN KEY (appointment_id)
 REFERENCES appointment(appointment_id),
 CONSTRAINT fk_veterinarian_id FOREIGN KEY (veterinarian_id)
 REFERENCES veterinarian(veterinarian_id),
 CONSTRAINT fk_medicine_id FOREIGN KEY (medicine_id)
 REFERENCES medicine(medicine_id),
 CONSTRAINT fk_cage_id FOREIGN KEY (cage_id)
 REFERENCES cage(cage_id)
);
